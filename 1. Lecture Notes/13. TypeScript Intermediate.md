---

## Generics

Generics let you write reusable code that works with multiple types while still preserving type safety. Think of a generic as a **type parameter** — a placeholder filled in when the function or class is used.

```ts
// Without generics: lose type information
function identity(value: any): any { return value; }

// With generics: type is preserved
function identity<T>(value: T): T { return value; }

const str = identity('hello');    // TypeScript infers T = string
const num = identity(42);         // TypeScript infers T = number

// Explicit type argument
identity<string>('hello');
```

### Generic functions

```ts
function first<T>(arr: T[]): T | undefined {
  return arr[0];
}

function zip<A, B>(a: A[], b: B[]): [A, B][] {
  return a.map((item, i) => [item, b[i]]);
}

zip([1, 2, 3], ['a', 'b', 'c']); // [number, string][]
```

### Generic interfaces

```ts
interface ApiResponse<T> {
  data: T;
  status: number;
  message: string;
}

interface PaginatedResponse<T> extends ApiResponse<T[]> {
  total: number;
  page: number;
  perPage: number;
}

type UserListResponse = PaginatedResponse<User>;
```

### Generic constraints

```ts
// T must have a `length` property
function logLength<T extends { length: number }>(value: T): void {
  console.log(value.length);
}

logLength('hello');      // OK
logLength([1, 2, 3]);    // OK
logLength(42);           // Error — number has no length

// Constrain T to be a key of another type
function getProperty<T, K extends keyof T>(obj: T, key: K): T[K] {
  return obj[key];
}

const name = getProperty(user, 'name');    // string
const age  = getProperty(user, 'age');     // number
getProperty(user, 'nonexistent');          // Error
```

## Utility Types

TypeScript provides built-in generic types for common transformations:

```ts
interface User {
  id: number;
  name: string;
  email: string;
  password: string;
}

// Partial — all properties become optional
type UserDraft = Partial<User>;

// Required — all properties become required
type RequiredUser = Required<User>;

// Readonly — all properties become readonly
type ImmutableUser = Readonly<User>;

// Pick — keep only the listed properties
type PublicUser = Pick<User, 'id' | 'name' | 'email'>;

// Omit — exclude the listed properties
type SafeUser = Omit<User, 'password'>;

// Record — creates an object type with specific keys and value type
type UserMap = Record<string, User>;
type Roles = Record<'admin' | 'editor' | 'viewer', string[]>;

// Exclude / Extract — filter union types
type Primitive = string | number | boolean | object;
type NonObject = Exclude<Primitive, object>;  // string | number | boolean
type Strings = Extract<Primitive, string>;    // string

// ReturnType — infer a function's return type
function getUser(): { id: number; name: string } { ... }
type UserReturn = ReturnType<typeof getUser>; // { id: number; name: string }

// Parameters — infer a function's parameter types as a tuple
type GetUserParams = Parameters<typeof getUser>;

// NonNullable — remove null and undefined
type Name = NonNullable<string | null | undefined>; // string
```

## Mapped Types

Mapped types transform every property in an existing type:

```ts
// Re-implementing Readonly
type MyReadonly<T> = {
  readonly [K in keyof T]: T[K];
};

// Re-implementing Partial
type MyPartial<T> = {
  [K in keyof T]?: T[K];
};

// Make all values nullable
type Nullable<T> = {
  [K in keyof T]: T[K] | null;
};

// Change all values to a different type
type Stringify<T> = {
  [K in keyof T]: string;
};

// Filter by value type (using `as` clause)
type StringOnly<T> = {
  [K in keyof T as T[K] extends string ? K : never]: T[K];
};
```

## Conditional Types

Types that evaluate differently based on a condition:

```ts
// Basic conditional type
type IsString<T> = T extends string ? true : false;

type A = IsString<string>;  // true
type B = IsString<number>;  // false

// Infer — extract a type from within another
type ReturnType<T> = T extends (...args: any[]) => infer R ? R : never;
type ArrayElement<T> = T extends (infer E)[] ? E : never;

type El = ArrayElement<string[]>;  // string

// Distributive conditional types
type ToArray<T> = T extends any ? T[] : never;
type StringOrNumberArray = ToArray<string | number>; // string[] | number[]
```

## Template Literal Types

```ts
type EventName = 'click' | 'focus' | 'blur';
type Handler = `on${Capitalize<EventName>}`;
// 'onClick' | 'onFocus' | 'onBlur'

type CSSProperty = 'margin' | 'padding';
type Side = 'Top' | 'Right' | 'Bottom' | 'Left';
type CSSShorthand = `${CSSProperty}${Side}`;
// 'marginTop' | 'marginRight' | ... | 'paddingLeft'

// Useful for typed event systems
type EventMap = {
  [K in Handler]: (event: MouseEvent) => void;
};
```

## Discriminated Unions

A pattern for modelling state machines — a union where each member has a common literal field (the "discriminant") that TypeScript uses to narrow:

```ts
type LoadingState = { status: 'loading' };
type SuccessState = { status: 'success'; data: User[] };
type ErrorState   = { status: 'error'; error: string };

type State = LoadingState | SuccessState | ErrorState;

function render(state: State) {
  switch (state.status) {
    case 'loading': return <Spinner />;
    case 'success': return <List users={state.data} />; // data is available here
    case 'error':   return <Error message={state.error} />;
  }
}
```

### Exhaustiveness checking

```ts
function assertNever(value: never): never {
  throw new Error(`Unhandled case: ${JSON.stringify(value)}`);
}

function render(state: State) {
  switch (state.status) {
    case 'loading': return '...';
    case 'success': return 'OK';
    // If you forget 'error', TypeScript errors here:
    default: return assertNever(state);
  }
}
```

## Type Guards

Custom functions that narrow types:

```ts
// Type predicate: `value is User` tells TypeScript the type after a truthy check
function isUser(value: unknown): value is User {
  return (
    typeof value === 'object' &&
    value !== null &&
    'id' in value &&
    'name' in value
  );
}

if (isUser(data)) {
  console.log(data.name); // TypeScript knows it's a User
}

// Assertion functions — throw if the condition is false, narrow after
function assertIsString(value: unknown): asserts value is string {
  if (typeof value !== 'string') throw new TypeError('Expected string');
}

assertIsString(input);
input.toUpperCase(); // TypeScript knows it's a string after this line
```

## Working with `unknown` and Safe Parsing

`unknown` is the type-safe alternative to `any`. You must narrow it before using it.

```ts
// Parsing API responses safely
async function fetchUser(id: number): Promise<User> {
  const res = await fetch(`/api/users/${id}`);
  const json: unknown = await res.json();

  if (!isUser(json)) {
    throw new Error('Invalid user data from API');
  }

  return json; // TypeScript is now confident this is a User
}
```

## Strict Mode Settings

`"strict": true` enables a bundle of checks. Understanding what each one does:

| Flag | What it prevents |
| ---- | ---------------- |
| `strictNullChecks` | Accessing properties on `null`/`undefined` |
| `noImplicitAny` | Variables without explicit types defaulting to `any` |
| `strictFunctionTypes` | Unsafe function parameter assignments |
| `strictPropertyInitialization` | Class properties used before assignment |
| `noUncheckedIndexedAccess` | Array access without checking for `undefined` |
| `exactOptionalPropertyTypes` | Assigning `undefined` to optional vs absent properties |

Enable `strict: true` plus `noUncheckedIndexedAccess` and `exactOptionalPropertyTypes` for the strongest guarantees.
