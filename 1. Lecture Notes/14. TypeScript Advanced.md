# TypeScript Advanced

---

## The Type System in Depth

TypeScript uses a **structural type system** (also called "duck typing"). Two types are compatible if they have the same structure — the name doesn't matter.

```ts
interface Point2D { x: number; y: number; }
interface Coordinate { x: number; y: number; }

const p: Point2D = { x: 1, y: 2 };
const c: Coordinate = p; // OK — same structure

// This also means a type with MORE properties is assignable to one with fewer
interface Named { name: string; }
const user = { name: 'Alice', age: 28 };
const named: Named = user; // OK — user has all required properties
```

## Advanced Inference with `infer`

`infer` lets you extract types from within complex types during conditional type evaluation:

```ts
// Extract the element type from an array or Promise
type Awaited<T> = T extends Promise<infer R> ? Awaited<R> : T;
type Unpacked<T> = T extends (infer E)[] ? E : T extends Promise<infer E> ? E : T;

// Extract function parameter and return types
type FirstParam<T extends (...args: any) => any> =
  T extends (first: infer F, ...rest: any[]) => any ? F : never;

// Extract the type of values in an object
type ValueOf<T> = T[keyof T];
type UserValues = ValueOf<{ id: number; name: string }>; // number | string

// Infer from a tuple
type Head<T extends any[]> = T extends [infer H, ...any[]] ? H : never;
type Tail<T extends any[]> = T extends [any, ...infer T] ? T : never;
```

## Recursive Types

```ts
// JSON value type
type JSONValue =
  | string
  | number
  | boolean
  | null
  | JSONValue[]
  | { [key: string]: JSONValue };

// Deeply nested readonly
type DeepReadonly<T> = {
  readonly [K in keyof T]: T[K] extends object ? DeepReadonly<T[K]> : T[K];
};

// Deep partial
type DeepPartial<T> = {
  [K in keyof T]?: T[K] extends object ? DeepPartial<T[K]> : T[K];
};

// Path types (like lodash _.get)
type Path<T> = T extends object
  ? { [K in keyof T]: K extends string ? K | `${K}.${Path<T[K]>}` : never }[keyof T]
  : never;
```

## Declaration Files (`.d.ts`)

Declaration files contain only type information — no runtime code. They describe the shape of JavaScript libraries so TypeScript knows how to use them.

```ts
// mylib.d.ts
declare module 'mylib' {
  export function parse(input: string): ParseResult;
  export interface ParseResult {
    tokens: string[];
    errors: string[];
  }
  export const VERSION: string;
}

// Ambient declarations (no module boundary)
declare const __DEV__: boolean;
declare function require(module: string): any;

// Augmenting global types
declare global {
  interface Array<T> {
    first(): T | undefined;
  }
}
```

The `@types/` namespace on npm (e.g. `@types/node`, `@types/express`) provides declaration files for popular libraries.

## Variance

Variance describes how subtype relationships propagate through generic types:

```ts
// Covariant — subtype relationship preserved (output positions)
// If Dog extends Animal, then Producer<Dog> extends Producer<Animal>
type Producer<T> = () => T;

// Contravariant — subtype relationship flips (input positions)
// If Dog extends Animal, then Consumer<Animal> extends Consumer<Dog>
type Consumer<T> = (value: T) => void;

// Invariant — no subtype relationship (read and write positions)
type ReadWrite<T> = { value: T };

// TypeScript 4.7+ explicit variance annotations
interface Box<out T> { value: T }     // covariant
interface Writer<in T> { write(v: T): void } // contravariant
```

Understanding variance explains why function parameters are checked contravariantly and return types covariantly.

## Satisfies Operator

`satisfies` validates that a value matches a type without widening the inferred type:

```ts
const palette = {
  red: [255, 0, 0],
  green: '#00ff00',
  blue: [0, 0, 255],
} satisfies Record<string, string | number[]>;

// Without satisfies: palette.red has type string | number[]
// With satisfies: palette.red has type number[] (preserves the specific inferred type)
palette.red.forEach(c => c.toFixed()); // OK — TypeScript knows it's number[]
palette.green.toUpperCase();            // OK — TypeScript knows it's string
```

## Branded Types

Branded types prevent mixing up values of the same underlying type that represent different things:

```ts
type UserId    = number & { readonly __brand: 'UserId' };
type ProductId = number & { readonly __brand: 'ProductId' };

function createUserId(id: number): UserId {
  return id as UserId;
}

function getUser(id: UserId): User { ... }

const userId    = createUserId(1);
const productId = 42 as ProductId;

getUser(userId);    // OK
getUser(productId); // Error — ProductId is not UserId
getUser(42);        // Error — plain number is not UserId
```

## Decorators (Stage 3)

Decorators are functions that wrap classes, methods, or properties to add behaviour:

```ts
// Class decorator
function singleton<T extends { new(...args: any[]): {} }>(constructor: T) {
  let instance: T;
  return class extends constructor {
    constructor(...args: any[]) {
      if (instance) return instance;
      super(...args);
      instance = this as any;
    }
  };
}

@singleton
class DatabaseConnection {
  connect() { ... }
}

// Method decorator
function log(target: any, key: string, descriptor: PropertyDescriptor) {
  const original = descriptor.value;
  descriptor.value = function(...args: any[]) {
    console.log(`Calling ${key} with`, args);
    const result = original.apply(this, args);
    console.log(`${key} returned`, result);
    return result;
  };
  return descriptor;
}

class UserService {
  @log
  getUser(id: number) { ... }
}
```

## Advanced Patterns

### Builder pattern with generics

```ts
class QueryBuilder<T = {}> {
  private conditions: T;

  where<K extends string, V>(key: K, value: V): QueryBuilder<T & Record<K, V>> {
    return new QueryBuilder({ ...this.conditions, [key]: value });
  }

  build(): T { return this.conditions; }
}

const query = new QueryBuilder()
  .where('name', 'Alice')
  .where('age', 28)
  .build(); // { name: string; age: number }
```

### Opaque types with phantom type parameters

```ts
type Opaque<T, Brand> = T & { readonly _brand: Brand };

type Metres   = Opaque<number, 'Metres'>;
type Seconds  = Opaque<number, 'Seconds'>;

const distance = 100 as Metres;
const duration = 30  as Seconds;

function speed(d: Metres, t: Seconds): number {
  return d / t;
}

speed(distance, duration); // OK
speed(duration, distance); // Error — types are reversed
```

### Type-safe event emitter

```ts
type EventMap = {
  login: { userId: string; timestamp: Date };
  logout: { userId: string };
  error: { message: string; code: number };
};

class TypedEmitter<Events extends Record<string, any>> {
  on<K extends keyof Events>(event: K, listener: (data: Events[K]) => void): void {}
  emit<K extends keyof Events>(event: K, data: Events[K]): void {}
}

const emitter = new TypedEmitter<EventMap>();
emitter.on('login', ({ userId, timestamp }) => {}); // fully typed
emitter.emit('error', { message: 'fail', code: 500 });
```
