## Streams

A stream is an abstraction for processing data piece by piece (in **chunks**), rather than loading everything into memory at once. This is critical when working with large files, video, or real-time data.

Imagine downloading a 4GB video. If you load the entire file into RAM before doing anything with it, you'll run out of memory. With streams, you process the first chunk while the rest is still downloading.

```js
import { createReadStream, createWriteStream } from 'node:fs';
import { createGzip } from 'node:zlib';

// Compress a large file without loading it into memory
// Each chunk flows through: read → compress → write
createReadStream('large-file.txt')
  .pipe(createGzip())
  .pipe(createWriteStream('large-file.txt.gz'))
  .on('finish', () => console.log('Compressed!'));
```

### Stream types

| Type | Description | Example |
| ---- | ----------- | ------- |
| Readable | Source of data — you read from it | `fs.createReadStream`, HTTP request body |
| Writable | Destination — you write to it | `fs.createWriteStream`, HTTP response |
| Duplex | Both readable and writable | Network sockets |
| Transform | Reads, transforms, and outputs | `zlib.createGzip`, `crypto.createCipher` |

### Using `pipeline` (safer than `.pipe`)

```js
import { pipeline } from 'node:stream/promises';
import { createReadStream, createWriteStream } from 'node:fs';
import { createGzip } from 'node:zlib';

await pipeline(
  createReadStream('input.txt'),
  createGzip(),
  createWriteStream('output.gz'),
); // handles errors and cleanup automatically
```

## Worker Threads

Node.js is single-threaded, but you can use **Worker Threads** for CPU-intensive operations — image processing, data parsing, cryptography — to avoid blocking the event loop.

```js
// worker.js
import { workerData, parentPort } from 'node:worker_threads';

// This heavy computation runs in a separate thread
const result = heavyComputation(workerData.input);
parentPort.postMessage(result);
```

```js
// main.js
import { Worker } from 'node:worker_threads';

function runWorker(data) {
  return new Promise((resolve, reject) => {
    const worker = new Worker('./worker.js', { workerData: data });
    worker.on('message', resolve);
    worker.on('error', reject);
  });
}

const result = await runWorker({ input: largeDataset });
```

## Clustering

A single Node.js process uses one CPU core. **Clustering** spawns multiple processes (one per core), each handling requests independently. A master process distributes incoming connections.

```js
import cluster from 'node:cluster';
import { cpus } from 'node:os';

if (cluster.isPrimary) {
  const numCPUs = cpus().length;
  console.log(`Primary ${process.pid} is running — forking ${numCPUs} workers`);

  for (let i = 0; i < numCPUs; i++) {
    cluster.fork();
  }

  cluster.on('exit', (worker) => {
    console.log(`Worker ${worker.process.pid} died — restarting`);
    cluster.fork();
  });
} else {
  // Each worker runs the Express app
  import('./app.js');
  console.log(`Worker ${process.pid} started`);
}
```

In production, use **PM2** which handles clustering, process management, and zero-downtime restarts:

```bash
pm2 start app.js -i max     # -i max uses all CPU cores
pm2 logs                    # view logs
pm2 monit                   # CPU/memory monitor
```

## Authentication and JWT

JSON Web Tokens (JWTs) are a compact, self-contained way to securely transmit information between parties. A JWT contains a payload (user data), is signed with a secret, and can be verified without a database lookup.

```
eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9    ← header
.eyJ1c2VySWQiOiIxMjMifQ                    ← payload (base64)
.SflKxwRJSMeKKF2QT4fwpMeJf36POk6yJV_adQssw5c  ← signature
```

```js
import jwt from 'jsonwebtoken';
import bcrypt from 'bcryptjs';

const JWT_SECRET = process.env.JWT_SECRET;

// Hash a password before storing it (NEVER store plain text passwords)
const hashedPassword = await bcrypt.hash(plainPassword, 12); // 12 = cost factor

// Verify a password attempt against the stored hash
const isValid = await bcrypt.compare(attempt, hashedPassword);

// Create a JWT token
function createToken(userId) {
  return jwt.sign(
    { userId },          // payload
    JWT_SECRET,          // signing secret
    { expiresIn: '7d' }  // expiry
  );
}

// Verify and decode a token
function verifyToken(token) {
  return jwt.verify(token, JWT_SECRET); // throws if invalid or expired
}

// Auth flow
app.post('/login', async (req, res) => {
  const { email, password } = req.body;
  const user = await db.findUserByEmail(email);

  if (!user || !(await bcrypt.compare(password, user.hashedPassword))) {
    return res.status(401).json({ error: 'Invalid credentials' });
  }

  const token = createToken(user.id);
  res
    .cookie('token', token, { httpOnly: true, secure: true, sameSite: 'strict' })
    .json({ message: 'Logged in' });
});
```

## WebSockets with Socket.io

HTTP is request-response: the client asks, the server answers, and the connection closes. **WebSockets** maintain an open, persistent connection — the server can push data to the client at any time. Essential for chat apps, live updates, collaborative tools, and games.

```js
// server.js
import express from 'express';
import { createServer } from 'node:http';
import { Server } from 'socket.io';

const app = express();
const httpServer = createServer(app);
const io = new Server(httpServer, {
  cors: { origin: 'http://localhost:3000' },
});

io.on('connection', (socket) => {
  console.log(`User connected: ${socket.id}`);

  socket.on('join-room', (roomId) => {
    socket.join(roomId);
    socket.to(roomId).emit('user-joined', socket.id);
  });

  socket.on('send-message', ({ roomId, message }) => {
    io.to(roomId).emit('new-message', { id: socket.id, message });
  });

  socket.on('disconnect', () => {
    console.log(`User disconnected: ${socket.id}`);
  });
});

httpServer.listen(4000);
```

```js
// client.js (browser)
import { io } from 'socket.io-client';

const socket = io('http://localhost:4000');

socket.emit('join-room', 'room-1');
socket.on('new-message', ({ id, message }) => {
  displayMessage(id, message);
});

function sendMessage(text) {
  socket.emit('send-message', { roomId: 'room-1', message: text });
}
```

## Database Connection Pooling

Opening a new database connection for every request is expensive. A **connection pool** maintains a set of open connections that are reused across requests.

```js
// postgres with pg-pool
import pg from 'pg';

const pool = new pg.Pool({
  connectionString: process.env.DATABASE_URL,
  max: 20,              // maximum connections in pool
  idleTimeoutMillis: 30000,
  connectionTimeoutMillis: 2000,
});

// Pool automatically manages connections
async function getUser(id) {
  const result = await pool.query('SELECT * FROM users WHERE id = $1', [id]);
  return result.rows[0];
}
```

## File Uploads

```js
import multer from 'multer';
import sharp from 'sharp';
import { S3Client, PutObjectCommand } from '@aws-sdk/client-s3';

const upload = multer({ storage: multer.memoryStorage() });
const s3 = new S3Client({ region: 'us-east-1' });

app.post('/upload', upload.single('image'), async (req, res) => {
  if (!req.file) return res.status(400).json({ error: 'No file uploaded' });

  // Resize and convert the image
  const processed = await sharp(req.file.buffer)
    .resize(800, 600, { fit: 'inside' })
    .webp({ quality: 80 })
    .toBuffer();

  const key = `uploads/${Date.now()}-${req.file.originalname}.webp`;

  // Upload to S3
  await s3.send(new PutObjectCommand({
    Bucket: process.env.S3_BUCKET,
    Key: key,
    Body: processed,
    ContentType: 'image/webp',
  }));

  res.json({ url: `https://${process.env.S3_BUCKET}.s3.amazonaws.com/${key}` });
});
```

## Email

```js
import nodemailer from 'nodemailer';

const transporter = nodemailer.createTransport({
  host: 'smtp.resend.com', // or use SendGrid, Mailgun, SES
  port: 465,
  secure: true,
  auth: { user: 'resend', pass: process.env.RESEND_API_KEY },
});

async function sendWelcomeEmail(to, name) {
  await transporter.sendMail({
    from: '"My App" <hello@myapp.com>',
    to,
    subject: `Welcome, ${name}!`,
    text: `Hi ${name}, welcome to My App!`,
    html: `<h1>Welcome, ${name}!</h1><p>Thanks for signing up.</p>`,
  });
}
```

## Testing APIs with Vitest and Supertest

```js
import { describe, it, expect, beforeAll, afterAll } from 'vitest';
import request from 'supertest';
import app from '../src/app.js';
import { db } from '../src/db.js';

describe('Users API', () => {
  beforeAll(async () => {
    await db.migrate.latest();
    await db.seed.run();
  });

  afterAll(async () => {
    await db.destroy();
  });

  it('GET /users returns all users', async () => {
    const res = await request(app).get('/users');
    expect(res.status).toBe(200);
    expect(Array.isArray(res.body)).toBe(true);
  });

  it('POST /users creates a user', async () => {
    const res = await request(app)
      .post('/users')
      .send({ name: 'Alice', email: 'alice@test.com' });

    expect(res.status).toBe(201);
    expect(res.body.name).toBe('Alice');
  });
});
```
